                        { 
                            int tMax;
                            int climateId = 0;
                            using (EntityContext db = new EntityContext())
                            {
                                var calcHeatsources = db.Heatsources.Where(h => h.HeatsourceName =
                                = heatsource.HeatsourceName);

                                 climateId = h.ClimateId;
